/* WAP to find power of a number*/

  #include <math.h>
#include <stdio.h>

int main() {
    double base, exp, result;
    printf("Enter a base number: ");
    scanf("%lf", &base);
    printf("Enter an exponent: ");
    scanf("%lf", &exp);

    // calculates the power
    result = pow(base, exp);

    printf("%.1lf^%.1lf = %.2lf", base, exp, result);
    return 0;
}
-----------------------------------

SOLUTION
Enter a base number: 5
Enter an exponent: 3
5.0^3.0 = 125.00
